how to build #

bash
git clone https://github.com/rbookr/vitepress ~/vitepress
cd ~/vitepress && pnpm install && pnpm build
# clone this repo
git clone https://github.com/rbookr/RBook ~/RBook
cd ~/Rbook && pnpm install
pnpm link ~/vitepress

# dev -> http://localhost:3000
pnpm dev

# build and serve
# -> http://localhost:5000
pnpm build
pnpm serve
